# ----------------------------------------------------------------------
# SYSTEM - PATHS
# ----------------------------------------------------------------------

# Ensure local binaries are available.
PATH=$PATH:$HOME/bin

# Pyenv path.
if [[ -d $HOME/.pyenv ]]; then
    export PATH="$HOME/.pyenv/bin:$PATH"        
    eval "$(pyenv init --path)"
    eval "$(pyenv virtualenv-init -)"
fi

# --------------------------------------------------------------------
# ESDOC - CREDENTIALS
# --------------------------------------------------------------------

if [[ -f $HOME/.esdoc/credentials ]]; then
    source $HOME/.esdoc/credentials
fi

# ----------------------------------------------------------------------
# ESDOC - PYESSV ARCHIVE
# ----------------------------------------------------------------------

if [[ -d /opt/pyessv-archive ]]; then
    export PYESSV_ARCHIVE_HOME=/opt/pyessv-archive
fi

# ----------------------------------------------------------------------
# ESDOC - CDF2CIM ARCHIVE
# ----------------------------------------------------------------------

if [[ -d /opt/cdf2cim-archive ]]; then
    export CDF2CIM_ARCHIVE_HOME=/opt/cdf2cim-archive
fi

# ----------------------------------------------------------------------
# ESDOC - CDF2CIM WEB SERVICE
# ----------------------------------------------------------------------

if [[ -d /opt/esdoc-errata-ws ]]; then
    export PYTHONPATH=/opt/esdoc-cdf2cim-ws:$PYTHONPATH
    source /opt/esdoc-cdf2cim-ws/sh/activate
fi
