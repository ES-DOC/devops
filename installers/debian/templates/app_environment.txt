# ----------------------------------------------------------------------
# PATHS
# ----------------------------------------------------------------------

# Ensure local binaries are available.
PATH=$PATH:$HOME/bin

# Pyenv path.
if [[ -d $HOME/.pyenv ]]; then
    export PATH="$HOME/.pyenv/bin:$PATH"        
    eval "$(pyenv init --path)"
    eval "$(pyenv virtualenv-init -)"
fi

# ----------------------------------------------------------------------
# POSTGRESQL
# ----------------------------------------------------------------------

export PATH=/usr/lib/postgresql/14/bin:$PATH
export PGDATA=/var/lib/pgsql/data

# --------------------------------------------------------------------
# ESDOC - CREDENTIALS
# --------------------------------------------------------------------

if [[ -f $HOME/.esdoc/credentials ]]; then
    source $HOME/.esdoc/credentials
fi

# ----------------------------------------------------------------------
# ESDOC - PYESSV ARCHIVE
# ----------------------------------------------------------------------

if [[ -d /opt/pyessv-archive ]]; then
    export PYESSV_ARCHIVE_HOME=/opt/pyessv-archive
fi

# ----------------------------------------------------------------------
# ESDOC - WEB SERVICES
# ----------------------------------------------------------------------

if [[ -d /opt/esdoc-cdf2cim-ws ]]; then
    export PYTHONPATH=/opt/esdoc-cdf2cim-ws:$PYTHONPATH
    source /opt/esdoc-cdf2cim-ws/sh/activate
fi

if [[ -d /opt/esdoc-documentation-ws ]]; then
    export PYTHONPATH=/opt/esdoc-documentation-ws:$PYTHONPATH
    source /opt/esdoc-documentation-ws/sh/activate
    export ERRATA_DB_USER=esdoc
    export ERRATA_DB_NAME=esdoc_errata
fi

if [[ -d /opt/esdoc-errata-ws ]]; then
    export PYTHONPATH=/opt/esdoc-errata-ws:$PYTHONPATH
    source /opt/esdoc-errata-ws/sh/activate
    export ERRATA_DB_USER=esdoc
    export ERRATA_DB_NAME=esdoc_errata
fi

if [[ -d /opt/pyessv-ws ]]; then
    export PYTHONPATH=/opt/pyessv-ws:$PYTHONPATH
    source /opt/pyessv-ws/sh/activate
fi
